CREATE TABLE Alunos (
    RA decimal(5) PRIMARY KEY,
    Nome varchar(40)
)

CREATE TABLE Disciplinas (
    Cod decimal(3) PRIMARY KEY,
    Nome varchar(40)
)

CREATE TABLE Matriculas (
    RA decimal(5),
    Cod decimal(3),
    PRIMARY KEY (RA,Cod),
    FOREIGN KEY (RA) REFERENCES Alunos(RA),
    FOREIGN KEY (Cod) REFERENCES Disciplinas(cod)
)

CREATE TABLE Resultados (
    RA decimal(5),
    Cod decimal(3),
    Nota decimal(4,1) NOT NULL,
    Freq decimal(5,1) NOT NULL,
    PRIMARY KEY (RA,Cod),
    FOREIGN KEY (RA) REFERENCES Alunos(RA),
    FOREIGN KEY (Cod) REFERENCES Disciplinas(cod)
)
